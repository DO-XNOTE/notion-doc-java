OK，来做一下环境的规划。我们把项目的环境分成四类一是 dev，表示开发环境，用来让开发人员在本机调试。二是 test，表示测试环境，测试环境部署在 K8S 上。三是 pre ，表示预发环境。预发环境是生产环境的一个预备环境，这个环境我们会尽量的让它和生产环境保持一致，这环境也会部署在 K8S 上。

最后是prod，表示正式环境，也会部署到 K8S 上。不同的环境之间，我们用 K8S 的 namespace 进行隔离，创建 3 个 namespace kubectl create 多 NS test，three，加 od。至于 dev，由于是开发人员本机的调试环境，所以我们就不去创建 namespace 了。

```xml
<plugin>
  <groupId>org.apache.maven.plugins</groupId>
  <artifactId>maven-release-plugin</artifactId>
  <configuration>
      <!-- 是否要自动设置子模块的版本号 -->
      <autoVersionSubmodules>true</autoVersionSubmodules>
      <!-- 发布时打的git tag的命名格式 -->
      <tagNameFormat>v@{project.version}</tagNameFormat>
      <!-- 设置运行的参数 -->
      <arguments>-DskipTests</arguments>
  </configuration>
</plugin>

-----------------------------------------------------------------

<distributionManagement>
    <!-- 指定用来发布非正式版本依赖的仓库是什么 -->
    <snapshotRepository>
        <id>maven-snapshots</id>

        <url>http://localhost:8081/repository/maven-snapshots/</url>
    </snapshotRepository>

    <repository>
        <id>maven-releases</id>
        <name>maven releases repo</name>
        <url>http://localhost:8081/repository/maven-releases/</url>
    </repository>
</distributionManagement>
```

```shell
pipeline {
    agent any

    stages {
        stage(' Test ') {
            steps {
                 sh "mvn test"
               }
            }
        stage(' Build ') {
            steps {
               script {
                    String imageName = "jenkins-ci-test-project"
                    String tag = "1.0.0-SNAPSHOT-$BUILD_NUMBER"
                    # docker的私服地址
                    String registry = "192.168.124.2:8888"
                    sh " " "
mvn clean install -DskipTests
mvn deploy
docker build -t $imageName:$tag  .

docker tag $imageName:$tag $registry/itmuch/$imageName:$tag
docker login -u XNOTE -p XNOTE@123 $registry
docker push $registry/itmuch/$imageName:$tag
docker logout #registry
" " "
               }
            }
        }
        stage(' Deploy ') {
            steps {
                script {
                   String imageName = "jenkins-ci-test-project"
                   String tag = "1.0.0-SNAPSHOT-$BUILD_NUMBER"
                   // docker的私服地址
                   String registry = "192.168.124.2:8888"
                   sh " " "
kubectl apply -n test -f - <<EOF

apiVersion: v1
kind: Service
metadata:
  name: jenkins-ci-test-project
spec:
  selector:
    app: jenkins-ci-test-project
  ports:
    - port: 8089
      targetPort: 8989
  type: NodePort

---
apiVersion: apps/v1
kind: Deployment
metadata:
  name: jenkins-ci-test-project
  labels:
    app:
spec:
  replicas: 2
  template:
    metadata:
      name: jenkins-ci-test-project
      labels:
        app: jenkins-ci-test-project
    spec:
      containers:
        - name: jenkins-ci-test-project
          image: $registry/itmuch/$imageName:$tag
          imagePullPolicy: IfNotPresent
          env:
            - name: "--spring.profiles.active"
              value: "test"
      restartPolicy: Always
  selector:
    matchLabels:
      app: jenkins-ci-test-project

EOF
" " "
                }
            }
        }
    }
}
```
